DROP FUNCTION `fc_descencriptar`; CREATE DEFINER=`root`@`localhost` FUNCTION `fc_descencriptar`(`clave_codificada` VARCHAR(10) CHARSET utf8) RETURNS VARCHAR(10) CHARSET utf8 NOT DETERMINISTIC CONTAINS SQL SQL SECURITY DEFINER BEGIN DECLARE var VARCHAR(10); SET var = (SELECT decode(clave_codificada,255)); RETURN var; END 


DROP FUNCTION `fc_encriptar`; CREATE DEFINER=`root`@`localhost` FUNCTION `fc_encriptar`(`clave` VARCHAR(10) CHARSET utf8) RETURNS VARCHAR(10) CHARSET utf8 NOT DETERMINISTIC CONTAINS SQL SQL SECURITY DEFINER BEGIN DECLARE var VARCHAR(10); SET var = (SELECT encode(clave,255)); return var; END
